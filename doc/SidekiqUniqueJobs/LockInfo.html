def set(obj)
  return unless SidekiqUniqueJobs.config.lock_info
  raise InvalidArgument, "argument `obj` (#{obj}) needs to be a hash" unless obj.is_a?(Hash)

  json = dump_json(obj)
  @value = load_json(json)
  super(json)
  value
end
