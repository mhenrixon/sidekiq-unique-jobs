def call
  log_info { "Adding dead #{item[CLASS]} job #{item[JID]}" }

  if deadset_kill?
    deadset_kill
  else
    push_to_deadset
  end
end

def push_to_deadset
  redis do |conn|
    conn.multi do
      conn.zadd("dead", now_f, payload)
      conn.zremrangebyscore("dead", "-inf", now_f - Sidekiq::DeadSet.timeout)
      conn.zremrangebyrank("dead", 0, -Sidekiq::DeadSet.max_jobs)
    end
  end
end
