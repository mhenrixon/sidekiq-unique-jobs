def calculate
  timeout = default_worker_options[LOCK_TIMEOUT]
  timeout = default_lock_timeout if default_lock_timeout
  timeout = worker_options[LOCK_TIMEOUT] if worker_options.key?(LOCK_TIMEOUT)
  timeout
end
