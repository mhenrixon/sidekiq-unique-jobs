def update(time, result, ex)
  if result
    log_info("(#{time}) Execution successfully returned #{result}")
  elsif ex.is_a?(Concurrent::TimeoutError)
    log_warn("(#{time}) Execution timed out")
  else
    log_info("(#{time}) Cleanup failed with error #{ex.message}")
    log_error(ex)
  end
end
