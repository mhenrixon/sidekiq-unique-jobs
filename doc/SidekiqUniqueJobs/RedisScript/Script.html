def initialize(name:, root_path:)
  @name      = name
  @root_path = root_path
  @path      = root_path.join("#{name}.lua").to_s
  @source    = render_file
  @sha       = compiled_sha
  @call_count = 0
end

def load(conn)
  @sha =
    if conn.respond_to?(:namespace)
      conn.redis.script(:load, source)
    else
      conn.script(:load, source)
    end

  self
end
