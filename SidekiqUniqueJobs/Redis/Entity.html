def exist?
  redis do |conn|
    # TODO: Remove the if statement in the future
    value =
      if conn.respond_to?(:exists?)
        conn.exists?(key)
      else
        conn.exists(key)
      end

    return value if boolean?(value)

    value.to_i.positive?
  end
end
