def reaper_registered?
  redis do |conn|
    conn.get(UNIQUE_REAPER).to_i + drift_reaper_interval > current_timestamp
  end
end

def restart_if_dead
  return if reaper_registered?

  log_info("Reaper is considered dead. Starting new reaper instance")
  orphans_manager.start
end

  end
