def call
  if sidekiq_worker_class?
    log_info("Rescheduling #{item[LOCK_DIGEST]}")
    worker_class.perform_in(5, *item[ARGS])
  else
    log_warn("Skip rescheduling of #{item[LOCK_DIGEST]} because #{worker_class} is not a Sidekiq::Worker")
  end
end
